# Read in data, check out set size
df = file_to_df(datapath+'data.tsv')

# First rough data clean up
# Delete unused columns and lower column names
df = delete_invariant_features(df)
df = lower_column_names(df)

# Time series
# Fix format
df.transaction_date = df.transaction_date.apply(lambda x: str(x) if len(str(x)) == 8 else '0'+str(x))
# Transform to datetime
df.transaction_date = [datetime.strptime(str(x), '%m%d%Y') for x in df.transaction_date]
df['year'] = df.transaction_date.apply(lambda x: x.year)
df['month'] = df.transaction_date.apply(lambda x: x.month)

# Let us focus on the chemicals first
droplist = [x for x in df.columns if 'reporter' in x or 'buyer' in x]
df = df[set(df.columns)-set(droplist)]
print(df.columns)
